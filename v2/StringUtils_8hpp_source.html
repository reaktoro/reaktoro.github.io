124 inline auto split(const String& str, const String& delims, std::function<String(String)> transform) -> Strings
125 {
126  Strings words;
127  std::size_t start = 0, end = 0;
128  while(end != String::npos)
129  {
130  end = str.find_first_of(delims, start);
131  String word = str.substr(start, end - start);
132  if(word != "") words.push_back(transform ? transform(word) : word);
133  start = end + 1;
134  }
135  return words;
136 }

161 {

164  {
165  auto uniqueword = words[i];
