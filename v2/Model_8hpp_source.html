218 template<typename Result, typename... Args>
219 auto chain(const Vec<Model<Result(Args...)>>& models) -> Model<Result(Args...)>
220 {
221  using ResultRef = Ref<Result>;
222 
223  const auto evalfns = vectorize(models, RKT_LAMBDA(model, model.evaluatorFn()));
224  const auto serializerfns = vectorize(models, RKT_LAMBDA(model, model.serializerFn()));
225 
226  auto evalfn = [=](ResultRef res, const Args&... args)
227  {
228  for(auto i = 0; i < evalfns.size(); ++i)
229  evalfns[i](res, args...);
230  };
231 
232  auto serializerfn = [serializerfns]() -> yaml
233  {
234  yaml result;
235  for(auto i = 0; i < serializerfns.size(); ++i)
236  result.push_back(serializerfns[i]());
237  return result;
238  };
239 
240  Params params;
241  for(const auto& model : models)
242  for(const auto& param : model.params())
243  params.append(param);
244 
245  return Model<Result(Args...)>(evalfn, params);
246 }
