463 private:
468  template<bool use_ideal_activity_model>
469  auto _update(const real& T, const real& P, ArrayXrConstRef n)
470  {
471  _data.T = T;
472  _data.P = P;
473  _data.n = n;
474 
475  const auto R = universalGasConstant;
476 
477  auto& nsum = _data.nsum;
478  auto& x = _data.x;
479  auto& G0 = _data.G0;
480  auto& H0 = _data.H0;
481  auto& V0 = _data.V0;
482  auto& Cp0 = _data.Cp0;
483  auto& Cv0 = _data.Cv0;
484  auto& Vex = _data.Vex;
485  auto& VexT = _data.VexT;
486  auto& VexP = _data.VexP;
487  auto& Gex = _data.Gex;
488  auto& Hex = _data.Hex;
489  auto& Cpex = _data.Cpex;
490  auto& Cvex = _data.Cvex;
491  auto& ln_g = _data.ln_g;
492  auto& ln_a = _data.ln_a;
493  auto& u = _data.u;
494 
495  const auto& species = phase().species();
496  const auto size = species.size();
497 
498  assert( n.size() == size );
499  assert( G0.size() == size );
500  assert( H0.size() == size );
501  assert( V0.size() == size );
502  assert( Cp0.size() == size );
503  assert( Cv0.size() == size );
504  assert( ln_g.size() == size );
505  assert( ln_a.size() == size );
506  assert( u.size() == size );
507 
508  // Compute the standard thermodynamic properties of the species in the phase.
509  StandardThermoProps aux;
510  for(auto i = 0; i < size; ++i)
511  {
512  aux = species[i].props(T, P);
513  G0[i] = aux.G0;
514  H0[i] = aux.H0;
515  V0[i] = aux.V0;
516  Cp0[i] = aux.Cp0;
517  Cv0[i] = aux.Cv0;
518  }
519 
520  // Compute the activity properties of the phase.
521  nsum = n.sum();
522 
523  if(nsum == 0.0)
524  x = (size == 1) ? 1.0 : 0.0;
525  else x = n / nsum;
526 
527  // Ensure there are no zero mole fractions
528  error(x.minCoeff() == 0.0, "Could not compute the chemical properties of phase ",
529  phase().name(), " because it has one or more species with zero amounts.");
530 
531  Vec<Any> extra;
532  ActivityPropsRef aprops{ Vex, VexT, VexP, Gex, Hex, Cpex, Cvex, ln_g, ln_a };
533  ActivityArgs args{ T, P, x, extra };
534  ActivityPropsFn activity_props_fn = use_ideal_activity_model ?
535  phase().idealActivityPropsFn() : phase().activityPropsFn();
536 
537  if(nsum == 0.0) aprops = 0.0;
538  else activity_props_fn(aprops, args);
539 
540  // Compute the chemical potentials of the species
541  u = G0 + R*T*ln_a;
542  }
543 };
