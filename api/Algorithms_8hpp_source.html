220 template<typename T>
221 auto range(T first, T last, T step)
222 {
223  if(last <= first) return std::vector<T>{};
224  auto size = std::size_t((last - first - 1)/step) + 1;
225  std::vector<T> res(size);
226  for(auto i = 0; i < size; ++i)
227  res[i] = first + i*step;
228  return res;
229 }
