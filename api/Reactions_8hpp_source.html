108  template<typename T>
109  auto add(T const& item) -> void
110  {
111  static_assert(
112  isConvertible<T, Reaction> ||
113  isConvertible<T, GeneralReaction> ||
114  isConvertible<T, ReactionGenerator>);
115 
116  if constexpr(isConvertible<T, Reaction>)
117  {
118  reaction_generators.push_back([=](SpeciesList const& species) -> Vec<Reaction> { return { item }; }); // item is a Reaction object
119  }
120  else if constexpr(isConvertible<T, GeneralReaction>)
121  {
122  reaction_generators.push_back([=](SpeciesList const& species) -> Vec<Reaction> { return { item(species) }; }); // item is a GeneralReaction object; use operator()(SpeciesList) to convert to Reaction
123  }
124  else
125  {
126  reaction_generators.push_back(item); // item is already a ReactionGenerator
127  }
128  }
129 
132  auto convert(SpeciesList const& species) const -> Vec<Reaction>;
133 
134 private:
136  Vec<ReactionGenerator> reaction_generators;
137 
139  template<typename Arg, typename... Args>
140  auto addAux(const Arg& arg, const Args&... args) -> void
141  {
142  add(arg);
143  if constexpr (sizeof...(Args) > 0)
144  addAux(args...);
145  }
