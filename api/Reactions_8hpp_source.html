140  template<typename T>
141  auto add(T const& item) -> void
142  {
143  static_assert(
144  isConvertible<T, Reaction> ||
145  isConvertible<T, GeneralReaction> ||
146  isConvertible<T, ReactionGenerator>);
147 
148  if constexpr(isConvertible<T, Reaction>)
149  {
150  reaction_generators.push_back([=](ReactionGeneratorArgs args) -> Vec<Reaction> { return { item }; }); // item is a Reaction object
151  }
152  else if constexpr(isConvertible<T, GeneralReaction>)
153  {
154  reaction_generators.push_back([=](ReactionGeneratorArgs args) -> Vec<Reaction> { return { item(args) }; }); // item is a GeneralReaction object; use operator()(ReactionGeneratorArgs) to convert to Reaction
155  }
156  else
157  {
158  reaction_generators.push_back(item); // item is already a ReactionGenerator
159  errorif(!reaction_generators.back(), "Expecting an object in Reactions::add call that produces an initialized ReactionGenerator function object.");
160  }
161  }
162 
165  auto convert(ReactionGeneratorArgs args) const -> Vec<Reaction>;
166 
167 private:
169  Vec<ReactionGenerator> reaction_generators;
170 
172  template<typename Arg, typename... Args>
173  auto addAux(const Arg& arg, const Args&... args) -> void
174  {
175  add(arg);
176  if constexpr (sizeof...(Args) > 0)
177  addAux(args...);
178  }
179 };
