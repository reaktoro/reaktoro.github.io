278 template<typename... GenericPhases>
279 Phases(const Database& db, const GenericPhases&... gphases)
280 : Phases(db)
281 {
282 static_assert(sizeof...(gphases) > 0);
283 addAux(gphases...);
284 }

287 auto add(const GenericPhase& phase) -> void;
288
290 auto add(const GenericPhasesGenerator& generator) -> void;

293 auto database() const -> const Database&;

299 operator Vec<Phase>() const;

301private:
303 Database db;
304
306 Vec<GenericPhase> genericphases;
307
309 Vec<GenericPhasesGenerator> generators;
310
312 template<typename Arg, typename... Args>
313 auto addAux(const Arg& arg, const Args&... args) -> void
314 {
315 add(arg);
316 if constexpr (sizeof...(Args) > 0)
317 addAux(args...);
318 }

412class SolidPhase : public GenericPhase
413{
414public:
416 explicit SolidPhase(const StringList& species) : GenericPhase(species) { initialize(); }
417
419 auto initialize() -> void
420 {
421 String phasename;
422 auto i = 0;
423 for(auto&& name : species())
424 phasename += (i++ == 0) ? name : "-" + name;
425
426 setName("SolidPhase");
427 setStateOfMatter(StateOfMatter::Solid);
428 setAggregateState(AggregateState::Solid);
429 setAdditionalAggregateStates({AggregateState::CrystallineSolid});
430 setActivityModel(ActivityModelIdealSolution(StateOfMatter::Solid));
431 setIdealActivityModel(ActivityModelIdealSolution(StateOfMatter::Solid));
