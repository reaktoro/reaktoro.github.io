181template<typename T>
182auto range(T first, T last, T step)
183{
184 if(last <= first) return std::vector<T>{};
185 auto size = std::size_t((last - first - 1)/step) + 1;
186 std::vector<T> res(size);
187 for(auto i = 0; i < size; ++i)
188 res[i] = first + i*step;
189 return res;
190}
