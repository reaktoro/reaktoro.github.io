149  auto update(const real& T, const real& P)
150  {
151  // Check if this update call can be skipped if T, P conditions remain the same
152  if(T == mdata.T && P == mdata.P)
153  return;
154 
155  mdata.T = T;
156  mdata.P = P;
157 
158  auto& G0 = mdata.G0;
159  auto& H0 = mdata.H0;
160  auto& V0 = mdata.V0;
161  auto& VT0 = mdata.VT0;
162  auto& VP0 = mdata.VP0;
163  auto& Cp0 = mdata.Cp0;
164 
165  const auto& species = mphase.species();
166  const auto size = species.size();
167 
168  assert(G0.size() == size);
169  assert(H0.size() == size);
170  assert(V0.size() == size);
171  assert(Cp0.size() == size);
172 
173  // Compute the standard thermodynamic properties of the species in the phase.
174  StandardThermoProps aux;
175  for(auto i = 0; i < size; ++i)
176  {
177  const auto& standard_thermo_model = mphase.species(i).standardThermoModel();
178  aux = standard_thermo_model ? standard_thermo_model(T, P) : StandardThermoProps{};
179  G0[i] = aux.G0;
180  H0[i] = aux.H0;
181  V0[i] = aux.V0;
182  VT0[i] = aux.VT0;
183  VP0[i] = aux.VP0;
184  Cp0[i] = aux.Cp0;
185  }
186  }

192  auto update(const real& T, const real& P, Wrt<real&> wrtvar)
193  {
194  autodiff::seed(wrtvar);
195  update(T, P);
196  autodiff::unseed(wrtvar);
197  }

254  auto speciesStandardEntropies() const -> ArrayXr
255  {
256  return (mdata.H0 - mdata.G0)/mdata.T; // from G0 = H0 - T*S0
257  }
258 
260  auto speciesStandardInternalEnergies() const -> ArrayXr
261  {
262  return mdata.H0 - mdata.P * mdata.V0; // from H0 = U0 + P*V0
263  }

266  auto speciesStandardHelmholtzEnergies() const -> ArrayXr
267  {
268  return mdata.G0 - mdata.P * mdata.V0; // from A0 = U0 - T*S0 = (H0 - P*V0) + (G0 - H0) = G0 - P*V0
269  }
