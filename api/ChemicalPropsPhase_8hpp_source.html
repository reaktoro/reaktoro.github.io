599  template<bool use_ideal_activity_model>
600  auto _update(const real& T, const real& P, ArrayXrConstRef n, Map<String, Any>& extra)
601  {
602  mdata.T = T;
603  mdata.P = P;
604  mdata.n = n;
605 
606  const auto R = universalGasConstant;
607 
608  auto& nsum = mdata.nsum;
609  auto& msum = mdata.msum;
610  auto& x = mdata.x;
611  auto& G0 = mdata.G0;
612  auto& H0 = mdata.H0;
613  auto& V0 = mdata.V0;
614  auto& VT0 = mdata.VT0;
615  auto& VP0 = mdata.VP0;
616  auto& Cp0 = mdata.Cp0;
617  auto& Vex = mdata.Vex;
618  auto& VexT = mdata.VexT;
619  auto& VexP = mdata.VexP;
620  auto& Gex = mdata.Gex;
621  auto& Hex = mdata.Hex;
622  auto& Cpex = mdata.Cpex;
623  auto& ln_g = mdata.ln_g;
624  auto& ln_a = mdata.ln_a;
625  auto& u = mdata.u;
626 
627  const auto& species = phase().species();
628  const auto N = species.size();
629 
630  assert( n.size() == N );
631  assert( G0.size() == N );
632  assert( H0.size() == N );
633  assert( V0.size() == N );
634  assert( VT0.size() == N );
635  assert( VP0.size() == N );
636  assert( Cp0.size() == N );
637  assert( ln_g.size() == N );
638  assert( ln_a.size() == N );
639  assert( u.size() == N );
640 
641  // Compute the standard thermodynamic properties of the species in the phase.
642  StandardThermoProps aux;
643  for(auto i = 0; i < N; ++i)
644  {
645  aux = species[i].standardThermoProps(T, P);
646  G0[i] = aux.G0;
647  H0[i] = aux.H0;
648  V0[i] = aux.V0;
649  VT0[i] = aux.VT0;
650  VP0[i] = aux.VP0;
651  Cp0[i] = aux.Cp0;
652  }
653 
654  // Compute the amount of the phase
655  nsum = n.sum();
656 
657  // Compute the mass of the phase
658  msum = (n * mphase.speciesMolarMasses()).sum();
659 
660  // Compute the mole fractions of the species
661  if(nsum == 0.0)
662  x = (N == 1) ? 1.0 : 0.0;
663  else x = n / nsum;
664 
665  // Ensure there are no zero mole fractions
666  error(x.minCoeff() == 0.0, "Could not compute the chemical properties of phase ",
667  phase().name(), " because it has one or more species with zero amounts.");
668 
669  // Compute the activity properties of the phase
670  ActivityPropsRef aprops{ Vex, VexT, VexP, Gex, Hex, Cpex, ln_g, ln_a, extra };
671  ActivityArgs args{ T, P, x };
672  const ActivityModel& activity_model = use_ideal_activity_model ? // IMPORTANT: Use `const ActivityModel&` here instead of `ActivityModel`, otherwise a new model is constructed without cache, and so memoization will not take effect.
673  phase().idealActivityModel() : phase().activityModel();
674 
675  if(nsum == 0.0) aprops = 0.0;
676  else activity_model(aprops, args);
677 
678  // Compute the chemical potentials of the species
679  u = G0 + R*T*ln_a;
680  }
681 };
