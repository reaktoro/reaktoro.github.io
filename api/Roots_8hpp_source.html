44 template<typename T>
45 auto cardano(const T& b, const T& c, const T& d) -> CubicRoots<T>
46 {
47  using std::abs;
48  using std::acos;
49  using std::cbrt;
50  using std::cos;
51  using std::sqrt;
52  using std::complex;
53 
54  const auto xn = -b/3;
55  const auto yn = d + xn*(c + xn*(b + xn));
56 
57  const auto delta2 = (b*b - 3*c)/9;
58 
59  const auto u = yn*yn;
60  const auto v = 4*delta2*delta2*delta2;
61 
62  const auto discr = u - v;
63 
64  const auto eps = 100*std::numeric_limits<T>::epsilon();
65 
66  if(discr > eps) {
67  const auto sqrtdiscr = sqrt(discr);
68  const auto aux1 = cbrt( 0.5*(-yn + sqrtdiscr) );
69  const auto aux2 = cbrt( 0.5*(-yn - sqrtdiscr) );
70  const auto alpha = xn + aux1 + aux2;
71  const auto z1 = alpha - xn;
72  const auto aux3 = 0.5*sqrt(3*(z1*z1 - 4*delta2));
73  const auto aux4 = xn - 0.5*z1;
74  const auto beta = complex{aux4, -aux3};
75  const auto gamma = complex{aux4, aux3};
76  return {alpha, beta, gamma};
77  }
78  else if(discr < -eps) {
79  const auto pi = 3.14159265358979323846;
80  const auto delta = sqrt(delta2);
81  const auto h = 2*delta2*delta;
82  const auto theta = acos(-yn/h) / 3.0;
83  const auto alpha = xn + 2*delta*cos(theta);
84  const auto beta = xn + 2*delta*cos(2*pi/3 - theta);
85  const auto gamma = xn + 2*delta*cos(2*pi/3 + theta);
86  return {gamma, beta, alpha};
87  }
88  else {
89  const auto delta = cbrt( 0.5*yn );
90  const auto alpha = xn + delta;
91  const auto gamma = xn - 2*delta;
92  if(alpha < gamma) return {alpha, alpha, gamma};
93  else return {gamma, alpha, alpha};
94  }
95 }

103 template<typename T>
104 auto newton(const std::function<std::tuple<T, T>(const T&)>& f, const T& x0, const T& epsilon, std::size_t maxiter) -> std::tuple<T, std::size_t, bool>
105 {
106  using std::abs;
107  assert(epsilon > 0.0);
108  assert(maxiter > 0);
109  T x = x0;
110  for(auto i = 0; i < maxiter; ++i)
111  {
112  const auto [fx, dfx] = f(x);
113  assert(dfx != 0.0);
114  x -= fx/dfx;
115  if(abs(fx) < epsilon)
116  return { x, i, true };
117  }
118  return { x, maxiter, false };
119 }
