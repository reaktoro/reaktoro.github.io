31template<typename ArrayConstRef, typename ArrayRef>
32auto molalities(ArrayConstRef&& n, Index iH2O, ArrayRef&& m)
33{
34 const auto N = n.size();
35 assert(iH2O < N);
36 const auto kgH2O = n[iH2O] * waterMolarMass;
37 const auto nsum = n.sum();
38 if(N == 1) {
39 m[iH2O] = 1.0;
40 return;
41 }
42 if(kgH2O == 0.0 || nsum == 0.0) {
43 m.fill(0.0);
44 m[iH2O] = 1.0;
45 return;
46 }
47 m = n/kgH2O;
48 m[iH2O] = n[iH2O]/nsum;
49}

68template<typename ArrayConstRef, typename MatrixRef>
69auto molalitiesJacobian(ArrayConstRef&& n, Index iH2O, MatrixRef&& J)
70{
71 //-----------------------------------------------------------------------------------------------------
72 // == LATEX ==
73 //-----------------------------------------------------------------------------------------------------
74 // \frac{\partial m_{i}}{\partial n_{j}}=m_{i}\left(\frac{\delta_{ij}}{n_{i}}-\frac{\delta_{jw}}{n_{w}}\right)
75 // \frac{\partial x_{w}}{\partial n_{j}}=\frac{1}{n_{\Sigma}}\left(\delta_{wj}-x_{w}\right)
76 //-----------------------------------------------------------------------------------------------------
77 using T = Decay<decltype(J(0, 0))>;
78 const auto N = n.size();
79 assert(iH2O < N);
80 assert(J.rows() == N);
81 assert(J.cols() == N);
82 J.fill(0.0);
83 const T nH2O = n[iH2O];
84 const T nsum = n.sum();
85 if(nH2O == 0.0 || nsum == 0.0)
86 return;
87 const T kgH2O = nH2O * waterMolarMass;
88 const T xH2O = nH2O/nsum;
89 const T kgH2Oinv = 1.0/kgH2O;
90 for(auto i = 0; i < N; ++i)
91 {
92 const T ni = n[i];
93 const T mi = ni/kgH2O;
94 J(i, i) = kgH2Oinv;
95 J(i, iH2O) -= mi/nH2O;
96 }
97 J.row(iH2O).array() = -xH2O/nsum;
98 J(iH2O, iH2O) += 1.0/nsum;
99}

118template<typename ArrayConstRef, typename MatrixRef>
119auto lnMolalitiesJacobian(ArrayConstRef&& n, Index iH2O, MatrixRef&& J) -> void
120{
121 //-----------------------------------------------------------------------------------------------------
122 // == LATEX ==
123 //-----------------------------------------------------------------------------------------------------
124 // \frac{\partial\ln m_{i}}{\partial n_{j}}=\frac{1}{m_{i}}\frac{\partial m_{i}}{\partial n_{j}}=\left(\frac{\delta_{ij}}{n_{i}}-\frac{\delta_{jw}}{n_{w}}\right)
125 // \frac{\partial\ln x_{w}}{\partial n_{j}}=\frac{\delta_{wj}}{n_{w}}-\frac{1}{n_{\Sigma}}
126 //-----------------------------------------------------------------------------------------------------
127 using T = Decay<decltype(J(0, 0))>;
128 const auto N = n.size();
129 assert(iH2O < N);
130 assert(J.rows() == N);
131 assert(J.cols() == N);
132 J.fill(0.0);
133 const T nH2O = n[iH2O];
134 const T nsum = n.sum();
135 if(nH2O == 0.0 || nsum == 0.0)
136 return;
137 for(auto i = 0; i < N; ++i)
138 {
139 const T ni = n[i];
140 J(i, i) = 1.0/ni;
141 J(i, iH2O) -= 1.0/nH2O;
142 }
143 J.row(iH2O).array() = -1.0/nsum;
144 J(iH2O, iH2O) += 1.0/nH2O;
145}

164template<typename ArrayConstRef, typename VectorRef>
165auto lnMolalitiesJacobianDiagonal(ArrayConstRef&& n, Index iH2O, VectorRef&& D) -> void
166{
167 //-----------------------------------------------------------------------------------------------------
168 // == LATEX ==
169 //-----------------------------------------------------------------------------------------------------
170 // \frac{\partial\ln m_{i}}{\partial n_{i}}=\frac{1}{n_{i}}-\frac{\delta_{iw}}{n_{w}}
171 //-----------------------------------------------------------------------------------------------------
172 using T = Decay<decltype(D[0])>;
173 const auto N = n.size();
174 assert(iH2O < N);
175 assert(D.rows() == N);
176 const T nH2O = n[iH2O];
177 const T nsum = n.sum();
178 if(nH2O == 0.0 || nsum == 0.0) {
179 D.fill(0.0);
180 return;
181 }
182 for(auto i = 0; i < N; ++i)
183 D[i] = 1.0/n[i];
184 D[iH2O] = 1.0/nH2O - 1.0/nsum;
185}
